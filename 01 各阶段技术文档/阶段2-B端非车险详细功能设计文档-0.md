## 一、模块概述

### 1.1 业务背景

非车险业务（寿险、健康险、意外险等）是保险中介公司的核心利润来源，具有件均保费高、佣金比例高、客户价值高的特点。本模块旨在为业务员提供一整套非车险展业工具，包括：

- **产品库管理**：浏览、搜索、对比非车险产品
- **保费试算**：快速计算保费和利益演示
- **计划书制作**：生成专业的保障方案计划书
- **客户管理CRM**：管理客户档案和跟进记录
- **订单管理**：跟踪订单状态
- **业绩统计**：可视化业绩数据
- **佣金管理**：查看佣金明细和提现

```java
@RestController
@RequestMapping("/insurance/product")
@Tag(name = "管理后台 - 产品管理")
public class ProductController {

    @Resource
    private ProductService productService;

    @GetMapping("/page")
    @Operation(summary = "获取产品分页列表")
    @PreAuthorize("@ss.hasPermission('insurance:product:query')")
    public CommonResult<PageResult<ProductRespVO>> getProductPage(
            @Valid ProductPageReqVO pageReqVO) {
        PageResult<ProductDO> pageResult = productService.getProductPage(pageReqVO);
        return success(ProductConvert.INSTANCE.convertPage(pageResult));
    }
}
```

**Service层：**
```java
@Service
public class ProductServiceImpl implements ProductService {

    @Resource
    private ProductMapper productMapper;

    @Override
    public PageResult<ProductDO> getProductPage(ProductPageReqVO pageReqVO) {
        // 构建查询条件
        LambdaQueryWrapperX<ProductDO> query = new LambdaQueryWrapperX<ProductDO>()
            .eqIfPresent(ProductDO::getCategoryId, pageReqVO.getCategoryId())
            .eqIfPresent(ProductDO::getInsuranceCompanyId, pageReqVO.getCompanyId())
            .eqIfPresent(ProductDO::getStatus, pageReqVO.getStatus())
            .likeIfPresent(ProductDO::getProductName, pageReqVO.getProductName())
            .orderByDesc(ProductDO::getSortOrder)
            .orderByDesc(ProductDO::getCreatedTime);
        
        return productMapper.selectPage(pageReqVO, query);
    }
}
```

```java
@Service
public class ProductCompareService {

    public CompareResultVO compareProducts(List<Long> productIds) {
        // 1. 查询产品详情
        List<ProductDO> products = productMapper.selectBatchIds(productIds);
        
        // 2. 提取对比维度
        List<CompareDimension> dimensions = extractDimensions(products);
        
        // 3. 构建对比矩阵
        List<List<String>> matrix = buildCompareMatrix(products, dimensions);
        
        // 4. 标记差异项
        markDifferences(matrix);
        
        return CompareResultVO.builder()
            .products(products)
            .dimensions(dimensions)
            .matrix(matrix)
            .build();
    }
    
    private List<CompareDimension> extractDimensions(List<ProductDO> products) {
        // 定义对比维度
        return Arrays.asList(
            new CompareDimension("保障期限", "coveragePeriod"),
            new CompareDimension("缴费期限", "paymentPeriod"),
            new CompareDimension("投保年龄", "ageRange"),
            new CompareDimension("保额范围", "amountRange"),
            new CompareDimension("重疾保障", "criticalIllness"),
            new CompareDimension("轻症保障", "minorIllness"),
            new CompareDimension("身故责任", "death"),
            new CompareDimension("保费豁免", "premiumWaiver")
        );
    }
    
    private void markDifferences(List<List<String>> matrix) {
        // 对每一行进行差异标记
        for (List<String> row : matrix) {
            Set<String> uniqueValues = new HashSet<>(row);
            if (uniqueValues.size() > 1) {
                // 存在差异，高亮显示
                row.add("HAS_DIFF");
            }
        }
    }
}
```

```groovy
def calculatePremium(params) {
    def age = params.age
    def gender = params.gender  // 1-男 2-女
    def amount = params.amount  // 保额(万元)
    def paymentYears = params.paymentYears  // 缴费年限

    // 基础费率查询
    def baseRate = getRateByAge(age, gender)

    def paymentFactor = getPaymentFactor(paymentYears)
    
    // 保障期系数
    def coverageFactor = getCoverageFactor(coverageYears)
    
    // 职业系数
    def occupationFactor = getOccupationFactor(params.occupationClass)
    
    // 计算年缴保费
    def annualPremium = amount * baseRate * paymentFactor * coverageFactor * occupationFactor
    
    return [
        annualPremium: annualPremium,
        totalPremium: annualPremium * paymentYears
    ]
}

        [age: 11..20, male: 0.0025, female: 0.0022],
        [age: 21..30, male: 0.0042, female: 0.0038],
        // ... 更多分段
    ]
```

```java
        params.put("paymentYears", reqVO.getPaymentYears());

        params.put("occupationClass", reqVO.getOccupationClass());

        // 3. 执行Groovy脚本
        String formula = product.getPremiumFormula();
        Map<String, Object> result = executeGroovyScript(formula, params);

        // 4. 封装返回结果
        return QuotationResultVO.builder()
            .productId(product.getId())

            .annualPremium(new BigDecimal(result.get("annualPremium").toString()))
            .totalPremium(new BigDecimal(result.get("totalPremium").toString()))
            .amount(reqVO.getAmount())
            .paymentPeriod(reqVO.getPaymentYears() + "年")
            .build();
    }

    /**
     * 执行Groovy脚本
     */
    private Map<String, Object> executeGroovyScript(String script, Map<String, Object> params) {

            GroovyShell shell = new GroovyShell();
            Script groovyScript = shell.parse(script);
            
            // 设置变量绑定
            Binding binding = new Binding();
            binding.setVariable("params", params);
            groovyScript.setBinding(binding);
            
            // 执行脚本
            Object result = groovyScript.run();
            
            return (Map<String, Object>) result;
        } catch (Exception e) {
            log.error("Groovy脚本执行失败: {}", e.getMessage(), e);
            throw new ServiceException("保费计算失败");
        }
    }
```

```java
@Service
public class CashValueService {

    /**
     * 生成现金价值表
     */
    public List<CashValueYearVO> generateCashValueTable(CashValueReqVO reqVO) {
        // 1. 获取产品的现金价值系数表
        ProductDO product = productMapper.selectById(reqVO.getProductId());
        JSONArray coeffTable = JSON.parseArray(product.getCashValueTable());
        
        List<CashValueYearVO> result = new ArrayList<>();
        
        BigDecimal annualPremium = reqVO.getAnnualPremium();
        BigDecimal cumulativePremium = BigDecimal.ZERO;
        
        // 2. 逐年计算现金价值
        for (int year = 1; year <= 100; year++) {
            if (year <= reqVO.getPaymentYears()) {
                cumulativePremium = cumulativePremium.add(annualPremium);
            }
            
            // 查找对应年份的系数
            BigDecimal coeff = findCoefficientByYear(coeffTable, year);
            
            // 计算当年现金价值
            BigDecimal cashValue = annualPremium
                .multiply(new BigDecimal(reqVO.getPaymentYears()))
                .multiply(coeff);
            
            // 计算IRR (内部收益率)
            BigDecimal irr = calculateIRR(cumulativePremium, cashValue, year);
            
            CashValueYearVO yearVO = CashValueYearVO.builder()
                .year(year)
                .age(reqVO.getInsuredAge() + year)
                .cumulativePremium(cumulativePremium)
                .cashValue(cashValue)
                .irr(irr)
                .build();
            
            result.add(yearVO);
            
            // 若现金价值为0则终止
            if (cashValue.compareTo(BigDecimal.ZERO) == 0 && year > 20) {
                break;
            }
        }
        
        return result;
    }
    
    /**
     * 计算IRR (简化算法)
     */
    private BigDecimal calculateIRR(BigDecimal totalIn, BigDecimal totalOut, int years) {
        if (totalIn.compareTo(BigDecimal.ZERO) == 0 || years == 0) {
            return BigDecimal.ZERO;
        }
        
        // IRR = (totalOut / totalIn) ^ (1/years) - 1
        double ratio = totalOut.divide(totalIn, 6, RoundingMode.HALF_UP).doubleValue();
        double irr = Math.pow(ratio, 1.0 / years) - 1;
        
        return new BigDecimal(irr * 100).setScale(2, RoundingMode.HALF_UP);
    }
}
```

```java
public class ProposalPdfService {

    @Resource
    private OssClient ossClient;

    @Resource
    private ProposalMapper proposalMapper;

    /**
     * 生成计划书PDF
     */
    public String generateProposalPdf(Long proposalId) {

        // 2. 构建HTML内容

        try {

    private byte[] convertHtmlToPdf(String htmlContent) {
```

```java
@Service
public class ShortLinkService {

    @Resource
    private RedisTemplate<String, String> redisTemplate;

    private static final String SHORT_LINK_PREFIX = "proposal:link:";
    private static final String BASE_URL = "https://m.yourcompany.com/h5/proposal/";

    /**
     * 生成短链
     */
    public String generateShortLink(Long proposalId) {
        // 1. 生成6位随机码
        String shortCode = RandomStringUtils.randomAlphanumeric(6);
        
        // 2. 存入Redis (7天过期)
        String key = SHORT_LINK_PREFIX + shortCode;
        redisTemplate.opsForValue().set(key, proposalId.toString(), 7, TimeUnit.DAYS);
        
        // 3. 返回完整URL
        return BASE_URL + shortCode;
    }
    
    /**
     * 解析短链
     */
    public Long parseShortLink(String shortCode) {
        String key = SHORT_LINK_PREFIX + shortCode;
        String proposalId = redisTemplate.opsForValue().get(key);
        
        if (StrUtil.isBlank(proposalId)) {
            throw new ServiceException("链接已过期或不存在");
        }
        
        return Long.parseLong(proposalId);
    }
}
```

```java
     */
    private Integer calculateValueScore(CustomerDO customer, List<OrderDO> orders) {
        int score = 0;
        
        // 保单数量 (最高30分)
        score += Math.min(orders.size() * 10, 30);
        
        // 累计保费 (最高40分)
        BigDecimal totalPremium = orders.stream()
            .map(OrderDO::getPremium)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
        score += Math.min(totalPremium.divide(new BigDecimal(10000), 0, RoundingMode.DOWN).intValue() * 5, 40);
        
        // 客户年龄 (最高15分)
        if (customer.getAge() >= 30 && customer.getAge() <= 45) {
            score += 15;  // 黄金年龄段
        } else if (customer.getAge() >= 25 && customer.getAge() < 30) {
            score += 10;
        }
        
        // 职业类别 (最高15分)
        if (customer.getOccupationClass() <= 2) {
            score += 15;
        } else if (customer.getOccupationClass() <= 4) {
            score += 10;
        }
        
        return Math.min(score, 100);
    }
```

```java
@Service
@Slf4j
public class CustomerImportService {

    @Resource
    private CustomerMapper customerMapper;

    /**
     * 导入客户
     */
    @Transactional(rollbackFor = Exception.class)
    public CustomerImportRespVO importCustomers(
            MultipartFile file, 
            Long agentId) {
        
        try {
            // 1. 读取Excel
            List<CustomerImportExcelVO> list = EasyExcel.read(file.getInputStream())
                .head(CustomerImportExcelVO.class)
                .sheet()
                .doReadSync();
            
            log.info("读取Excel记录数: {}", list.size());
            
            // 2. 数据校验与去重
            List<CustomerDO> validList = new ArrayList<>();
            List<String> errorList = new ArrayList<>();
            Set<String> mobileSet = new HashSet<>();
            
            for (int i = 0; i < list.size(); i++) {
                CustomerImportExcelVO vo = list.get(i);
                int rowNum = i + 2;  // Excel行号(从第2行开始)
                
                try {
                    // 校验必填项
                    if (StrUtil.isBlank(vo.getName())) {
                        errorList.add(String.format("第%d行: 客户姓名不能为空", rowNum));
                        continue;
                    }
                    
                    // 检查重复手机号
                    if (mobileSet.contains(vo.getMobile())) {
                        errorList.add(String.format("第%d行: 手机号重复", rowNum));
                        continue;
                    }
                    mobileSet.add(vo.getMobile());
                    
                    // 检查数据库是否已存在
                    Long count = customerMapper.selectCount(
                        new LambdaQueryWrapperX<CustomerDO>()
                            .eq(CustomerDO::getMobile, vo.getMobile())
                            .eq(CustomerDO::getAgentId, agentId)
                    );
                    if (count > 0) {
                        errorList.add(String.format("第%d行: 该客户已存在", rowNum));
                        continue;
                    }
                    
                    // 转换为DO对象
                    CustomerDO customer = convertToCustomerDO(vo, agentId);
                    validList.add(customer);
                    
                } catch (Exception e) {
                    errorList.add(String.format("第%d行: %s", rowNum, e.getMessage()));
                }
            }
            
            // 3. 批量插入
            if (!validList.isEmpty()) {
                customerMapper.insertBatch(validList);
            }
            
            // 4. 返回导入结果
            return CustomerImportRespVO.builder()
                .totalCount(list.size())
                .successCount(validList.size())
                .failCount(errorList.size())
                .errorList(errorList)
                .build();
            
        } catch (Exception e) {
            log.error("客户导入失败", e);
            throw new ServiceException("导入失败: " + e.getMessage());
        }
    }
    
    private CustomerDO convertToCustomerDO(CustomerImportExcelVO vo, Long agentId) {
        CustomerDO customer = new CustomerDO();
        customer.setCustomerNo(generateCustomerNo());
        customer.setAgentId(agentId);
        customer.setName(vo.getName());
        customer.setMobile(AesUtils.encrypt(vo.getMobile()));  // 手机号加密
        customer.setGender("男".equals(vo.getGenderText()) ? 1 : 2);
        customer.setBirthday(vo.getBirthday());
        
        // 计算年龄
        if (vo.getBirthday() != null) {
            customer.setAge(Period.between(vo.getBirthday(), LocalDate.now()).getYears());
        }
        
        customer.setOccupation(vo.getOccupation());
        customer.setCity(vo.getCity());
        customer.setRemark(vo.getRemark());
        customer.setSource("Excel导入");
        
        return customer;
    }
    
    private String generateCustomerNo() {
        return "C" + IdUtil.getSnowflakeNextIdStr();
    }
}
```

#### 8.1.1 后端统计Service

```java
@Service
public class PerformanceStatsService {

    @Resource
    private OrderMapper orderMapper;

    /**
     * 获取业绩看板数据
     */
    public PerformanceDashboardVO getDashboard(Long agentId) {
        LocalDateTime now = LocalDateTime.now();
        
        // 1. 今日业绩
        PerformanceVO today = calculatePerformance(
            agentId,
            now.toLocalDate().atStartOfDay(),
            now
        );
        
        // 2. 本月业绩
        LocalDateTime monthStart = now.withDayOfMonth(1).withHour(0).withMinute(0).withSecond(0);
        PerformanceVO thisMonth = calculatePerformance(agentId, monthStart, now);
        
        // 3. 累计业绩
        PerformanceVO total = calculatePerformance(agentId, null, now);
        
        // 4. 本月趋势(每日数据)
        List<DailyPerformanceVO> trend = calculateMonthlyTrend(agentId, monthStart, now);
        
        return PerformanceDashboardVO.builder()
            .today(today)
            .thisMonth(thisMonth)
            .total(total)
            .trend(trend)
            .build();
    }
    
    private PerformanceVO calculatePerformance(
            Long agentId, 
            LocalDateTime startTime, 
            LocalDateTime endTime) {
        
        // 查询订单
        LambdaQueryWrapperX<OrderDO> query = new LambdaQueryWrapperX<OrderDO>()
            .eq(OrderDO::getAgentId, agentId)
            .eq(OrderDO::getOrderStatus, 3);  // 已承保
        
        if (startTime != null) {
            query.ge(OrderDO::getIssueTime, startTime);
        }
        if (endTime != null) {
            query.le(OrderDO::getIssueTime, endTime);
        }
        
        List<OrderDO> orders = orderMapper.selectList(query);
        
        // 统计
        BigDecimal totalPremium = orders.stream()
            .map(OrderDO::getPremium)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
        
        BigDecimal totalCommission = orders.stream()
            .map(OrderDO::getCommissionAmount)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
        
        return PerformanceVO.builder()
            .orderCount(orders.size())
            .totalPremium(totalPremium)
            .totalCommission(totalCommission)
            .avgPremium(orders.isEmpty() ? BigDecimal.ZERO : 
                totalPremium.divide(new BigDecimal(orders.size()), 2, RoundingMode.HALF_UP))
            .build();
    }
    
    /**
     * 计算月度趋势
     */
    private List<DailyPerformanceVO> calculateMonthlyTrend(
            Long agentId, 
            LocalDateTime monthStart, 
            LocalDateTime now) {
        
        List<DailyPerformanceVO> result = new ArrayList<>();
        LocalDate currentDate = monthStart.toLocalDate();
        LocalDate endDate = now.toLocalDate();
        
        while (!currentDate.isAfter(endDate)) {
            LocalDateTime dayStart = currentDate.atStartOfDay();
            LocalDateTime dayEnd = currentDate.plusDays(1).atStartOfDay();
            
            PerformanceVO dayPerf = calculatePerformance(agentId, dayStart, dayEnd);
            
            result.add(DailyPerformanceVO.builder()
                .date(currentDate.toString())
                .orderCount(dayPerf.getOrderCount())
                .premium(dayPerf.getTotalPremium())
                .commission(dayPerf.getTotalCommission())
                .build());
            
            currentDate = currentDate.plusDays(1);
        }
        
        return result;
    }
}
```

```java
@PostMapping("/withdrawal/apply")
@Operation(summary = "申请提现")
public CommonResult<Long> applyWithdrawal(@Valid @RequestBody WithdrawalApplyReqVO reqVO) {
    // 1. 查询可提现佣金
    BigDecimal availableAmount = commissionMapper.selectAvailableAmount(reqVO.getAgentId());
    
    if (availableAmount.compareTo(reqVO.getAmount()) < 0) {
        throw new ServiceException("可提现金额不足");
    }
    
    // 2. 创建提现记录
    WithdrawalDO withdrawal = new WithdrawalDO();
    withdrawal.setWithdrawalNo(generateWithdrawalNo());
    withdrawal.setAgentId(reqVO.getAgentId());
    withdrawal.setAmount(reqVO.getAmount());
    withdrawal.setBankName(reqVO.getBankName());
    withdrawal.setBankAccount(AesUtils.encrypt(reqVO.getBankAccount()));
    withdrawal.setAccountName(reqVO.getAccountName());
    withdrawal.setStatus(1);  // 审核中
    withdrawal.setApplyTime(LocalDateTime.now());
    
    withdrawalMapper.insert(withdrawal);
    
    // 3. 锁定对应佣金
    commissionMapper.lockCommissionForWithdrawal(
        reqVO.getAgentId(), 
        reqVO.getAmount(), 
        withdrawal.getId()
    );
    
    return success(withdrawal.getId());
}
```

## 十一、性能优化建议

### 11.1 Redis缓存策略

```java
@Service
public class ProductService {

    private static final String PRODUCT_CACHE_KEY = "product:detail:";
    private static final long CACHE_EXPIRE = 3600L;  // 1小时

    @Resource
    private RedisTemplate<String, Object> redisTemplate;

    public ProductDetailRespVO getProductDetail(Long id) {
        // 1. 查缓存
        String cacheKey = PRODUCT_CACHE_KEY + id;
        ProductDetailRespVO cached = (ProductDetailRespVO) redisTemplate.opsForValue().get(cacheKey);
        
        if (cached != null) {
            return cached;
        }
        
        // 2. 查数据库
        ProductDO product = productMapper.selectById(id);
        ProductDetailRespVO result = ProductConvert.INSTANCE.convertDetail(product);
        
        // 3. 写缓存
        redisTemplate.opsForValue().set(cacheKey, result, CACHE_EXPIRE, TimeUnit.SECONDS);
        
        return result;
    }
}
```

```java
// 使用游标分页而非offset分页
public PageResult<OrderDO> getOrderPageOptimized(OrderPageReqVO pageReqVO) {
    // 记录上次查询的最后一条记录ID
    Long lastId = pageReqVO.getLastId();
    
    LambdaQueryWrapperX<OrderDO> query = new LambdaQueryWrapperX<OrderDO>()
        .eq(OrderDO::getAgentId, pageReqVO.getAgentId())
        .gt(lastId != null, OrderDO::getId, lastId)  // 游标条件
        .orderByAsc(OrderDO::getId)
        .last("LIMIT " + pageReqVO.getPageSize());
    
    List<OrderDO> list = orderMapper.selectList(query);
    
    return new PageResult<>(list, null);  // 游标分页不返回total
}
```
